2.Servo

#include<Servo.h>
Servo servoBase;
void setup(){
  servoBase.attach(A1);
  servoBase.write(0);
  Serial.begin(9600);
 

}
void loop(){
  for (int i=0; i<=180; i+=10){
  servoBase.write(i);
  delay(1000);}

}

4.LDR

int sV(0);
void setup(){
pinMode(A0,INPUT);
  pinMode(0,OUTPUT);
  Serial.begin(9600);
}
void loop(){
  sV=analogRead(A0);
  Serial.println(sV);
  analogWrite(9,map(sV,0,1023,255,0));
  delay(1000);

}
